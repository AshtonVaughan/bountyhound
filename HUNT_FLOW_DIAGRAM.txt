│ HTTP POST /api/fuzz                                                      │

│ {                                                                         │
│   "job_id": "a3f2e1c9",                                                 │

│ }                                                                         │
│                                                                            │

│ ─────────────────────────                                               │

│ 2. Parses output                                                        │
│ 3. Creates FfufResult objects with found endpoints                      │

│ 5. Sets job.status = "completed"                                       │
│ 6. Auto-cleanup: removes old jobs (max 100 per tool)                   │
│                                                                            │
│ HTTP GET /api/fuzz/{job_id}                                             │
│ Returns current job state (agent polls this if needed):                 │
│ {                                                                         │
│   "job_id": "a3f2e1c9",                                                │
│   "status": "completed",                                               │
│   "results": [                                                         │
│     {"url": "target.com/admin", "status": 200, ...},                   │
│     {"url": "target.com/api", "status": 401, ...},                     │
│     ...                                                                │
│   ],                                                                    │
│   "total_results": 42                                                 │
│ }                                                                        │
